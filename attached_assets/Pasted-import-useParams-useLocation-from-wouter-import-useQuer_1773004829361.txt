import { useParams, useLocation } from "wouter";
import { useQuery } from "@tanstack/react-query";
import { Avatar, AvatarFallback } from "@/components/ui/avatar";
import { ArrowLeft } from "lucide-react";
import CommunityPost, { type CommunityPost as CommunityPostType } from "@/components/CommunityPost";
import type { CommunityPost as CommunityPostSchema } from "@shared/schema";

export default function Profile() {
  const { username } = useParams<{ username: string }>();
  const [, setLocation] = useLocation();

  const { data: posts = [], isLoading } = useQuery<CommunityPostSchema[]>({
    queryKey: ["/api/community"],
  });

  const decodedUsername = decodeURIComponent(username || "");

  const userPosts = (posts as any[]).filter(
    (p) => p.userName === decodedUsername
  );

  const formattedPosts: CommunityPostType[] = userPosts.map((post) => ({
    id: post.id,
    userName: post.userName,
    userAvatar: post.userAvatar || undefined,
    cigarName: post.cigarName,
    brand: post.brand || undefined,
    rating: post.rating,
    comment: post.comment || undefined,
    timestamp: new Date(post.timestamp),
    likes: post.likes,
    comments: post.comments,
    imageUrl: post.imageUrl || undefined,
  }));

  const totalLikes = userPosts.reduce((sum, p) => sum + (p.likes || 0), 0);
  const avgRating = userPosts.filter(p => p.rating > 0).length > 0
    ? (userPosts.filter(p => p.rating > 0).reduce((sum, p) => sum + p.rating, 0) / userPosts.filter(p => p.rating > 0).length).toFixed(1)
    : "—";

  const initials = decodedUsername.split(" ").map((n) => n[0]).join("").toUpperCase().slice(0, 2);

  return (
    <div className="min-h-screen bg-background">
      <div className="max-w-xl mx-auto px-6 py-10">

        <button
          onClick={() => setLocation("/community")}
          className="flex items-center gap-2 text-xs uppercase tracking-widest text-muted-foreground font-light mb-10 hover:text-foreground transition-colors"
        >
          <ArrowLeft className="w-3.5 h-3.5" />
          Back
        </button>

        <div className="flex items-end gap-6 mb-10 border-b border-border pb-8">
          <div className="p-0.5 rounded-full bg-gradient-to-tr from-primary to-amber-300">
            <Avatar className="w-16 h-16 border-2 border-background">
              <AvatarFallback className="text-lg font-light">{initials}</AvatarFallback>
            </Avatar>
          </div>
          <div className="flex-1">
            <p className="text-xs uppercase tracking-[0.3em] text-muted-foreground font-light mb-1">Aficionado</p>
            <h1 className="text-4xl font-serif font-light tracking-wide">{decodedUsername}</h1>
          </div>
        </div>

        <div className="grid grid-cols-3 gap-px bg-border mb-10">
          <div className="bg-background p-5 text-center">
            <p className="text-3xl font-serif font-light mb-1">{userPosts.length}</p>
            <p className="text-xs uppercase tracking-widest text-muted-foreground font-light">Posts</p>
          </div>
          <div className="bg-background p-5 text-center">
            <p className="text-3xl font-serif font-light mb-1">{totalLikes}</p>
            <p className="text-xs uppercase tracking-widest text-muted-foreground font-light">Likes</p>
          </div>
          <div className="bg-background p-5 text-center">
            <p className="text-3xl font-serif font-light mb-1">{avgRating}</p>
            <p className="text-xs uppercase tracking-widest text-muted-foreground font-light">Avg Rating</p>
          </div>
        </div>

        <div className="mb-6">
          <p className="text-xs uppercase tracking-[0.3em] text-muted-foreground font-light">Sessions</p>
        </div>

        {isLoading ? (
          <div className="space-y-4">
            {[1, 2, 3].map((i) => <div key={i} className="h-32 animate-pulse bg-muted/20" />)}
          </div>
        ) : formattedPosts.length === 0 ? (
          <div className="py-16 text-center border border-border">
            <p className="text-xs uppercase tracking-widest text-muted-foreground font-light">No posts yet</p>
          </div>
        ) : (
          <div>
            {formattedPosts.map((post) => (
              <CommunityPost key={post.id} post={post} />
            ))}
          </div>
        )}
      </div>
    </div>
  );
}
